// [index.html 내 관리자 로그인 함수] function checkAdmin() { const savedAdmin = JSON.parse(localStorage.getItem("assignedAdmin")); if (!savedAdmin) { // 1. 담당자가 지정되지 않은 최초 상태 const masterPw = prompt("담당자가 지정되지 않았습니다.\n마스터 비밀번호를 입력하세요."); if (masterPw === "1234") { alert("마스터 인증 성공! 담당자 지정 페이지로 이동합니다."); location.href = "admin_register.html"; // 담당자 등록 페이지 (새로 만들 것) } else if (masterPw) { alert("마스터 비밀번호가 틀렸습니다."); } } else { // 2. 담당자가 이미 지정된 상태 const adminId = prompt("관리자 사번을 입력하세요."); if (adminId === savedAdmin.id) { const adminPw = prompt("관리자 개인 비밀번호를 입력하세요."); if (adminPw === savedAdmin.pw) { location.href = "admin_main.html"; } else if (adminPw) { alert("비밀번호가 일치하지 않습니다."); } } else if (adminId) { alert("등록된 관리자 사번이 아닙니다."); } } }
