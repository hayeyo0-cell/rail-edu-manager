function handleLogin() { const team = document.getElementById('team-select').value; const id = document.getElementById('id-input').value; const pw = document.getElementById('pw-input').value; if(!team || !pw) { alert("소속과 비밀번호를 확인해주세요."); return; } if(currentMode === 'user') { // [기관사 로그인 로직 - 기존과 동일] const user = userDB[team].find(u => u.id === id && u.pw === pw); if(user) { localStorage.setItem("loginUser", JSON.stringify({ ...user, team: team })); window.location.href = "user_main.html"; } else { alert("정보가 틀렸습니다."); } } else { // [관리자 로그인 로직 - 새로운 방식] const savedAdmin = JSON.parse(localStorage.getItem(`adminConfig_${team}`)); if (!savedAdmin) { // 1. 등록된 관리자가 없을 때 (초기 비번 1234로 입장) if (pw === "1234") { alert("관리자 미등록 상태입니다. 관리자 등록 화면으로 이동합니다."); localStorage.setItem("tempAdminAccess", team); // 임시 권한 부여 window.location.href = "admin_register.html"; // 등록 페이지로 이동 } else { alert("비밀번호가 틀렸습니다. 초기 비밀번호는 1234입니다."); } } else { // 2. 관리자가 이미 등록되어 있을 때 (사번+비번 대조) if (id === savedAdmin.id && pw === savedAdmin.pw) { localStorage.setItem("loginAdmin", team); window.location.href = "admin_main.html"; } else { alert("등록된 관리자 정보와 일치하지 않습니다."); } } } }
