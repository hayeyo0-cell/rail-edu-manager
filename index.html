// --- [데이터베이스 영역] --- // 모든 팀에 기관사님의 정보를 복사해서 넣었습니다. const commonUser = {id: "20050552", name: "테스터", pw: "1234"}; const userDB = { "KS": [ commonUser ], "MY": [ commonUser ], "WB": [ commonUser ], "AS": [ commonUser ] }; // 관리자 비번도 모든 팀 1234로 통일 const adminDB = { "KS": "1234", "MY": "1234", "WB": "1234", "AS": "1234" }; let currentMode = 'user'; function setMode(mode) { currentMode = mode; document.getElementById('btn-user').classList.toggle('active', mode === 'user'); document.getElementById('btn-admin').classList.toggle('active', mode === 'admin'); document.getElementById('title').innerText = mode === 'user' ? '기관사 교육관리' : '관리자 교육관리'; document.getElementById('id-label').innerText = mode === 'user' ? '사번' : '관리자 ID'; } function handleLogin() { const team = document.getElementById('team-select').value; const id = document.getElementById('id-input').value; const pw = document.getElementById('pw-input').value; if(!team || !id || !pw) { alert("모든 정보를 입력해주세요."); return; } if(currentMode === 'user') { const teamUsers = userDB[team]; const user = teamUsers.find(u => u.id === id && u.pw === pw); if(user) { // 로그인 정보 저장 localStorage.setItem("loginUser", JSON.stringify({ ...user, team: team })); alert(`${team} 사업소 ${user.name}님 환영합니다!`); window.location.href = "user_main.html"; } else { alert("사번 또는 비밀번호가 틀렸습니다."); } } else { // 관리자 로그인: ID 상관없이 선택한 팀의 비번만 맞으면 통과 if(pw === adminDB[team]) { localStorage.setItem("loginAdmin", team); alert(`${team} 관리자 모드로 접속합니다.`); window.location.href = "admin_main.html"; } else { alert("관리자 비밀번호가 일치하지 않습니다."); } } }
