function handleFinalSave() { // 1. 값 가져오기 const newTeamPw = document.getElementById('newTeamPw').value; const newAdminId = document.getElementById('newAdminId').value; const newAdminPw = document.getElementById('newAdminPw').value; // 2. 일치 여부 최종 확인 const checks = ['TeamPw', 'AdminId', 'AdminPw']; for (let type of checks) { const original = document.getElementById('new' + type).value; const confirm = document.getElementById('confirm' + type).value; if (original === "" || original !== confirm) { alert('입력되지 않았거나 일치하지 않는 항목이 있습니다.'); return; } } if(confirm('설정 내용을 저장하시겠습니까?\n이후 로그인 시 변경된 정보로 접속해야 합니다.')) { // [핵심 저장 로직] // index.html에서 참조하는 이름과 똑같이 맞춰서 저장합니다. localStorage.setItem('teamPw', newTeamPw); // 팀 공용 비번 localStorage.setItem('adminId', newAdminId); // 관리자 사번 localStorage.setItem('adminPw', newAdminPw); // 관리자 비번 // 현재 관리자의 '소속'은 이미 등록 시점에 저장되어 있다고 가정합니다. // (만약 소속 선택 기능이 필요하면 여기에 추가하면 됩니다.) alert('모든 변경사항이 안전하게 저장되었습니다.\n보안을 위해 다시 로그인해 주세요.'); location.href='index.html'; // 저장 후 다시 로그인하게 유도 } }
